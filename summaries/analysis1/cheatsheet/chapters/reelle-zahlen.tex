\section{Reelle und komplexe Zahlen}

\Title{Axiome der Reelle Zahlen} \\
$\R$ ist ein komm., angeordneter KûÑrper, der ordnungsvollstûÊndig ist.

\begin{multicols}{2}
\tiny
\begin{itemize}
\itemsep0em 
	\item[(A)]{\textbf{Axiome der Addition} $x,y \in \R \Rightarrow x + y \in \R$}
	\begin{itemize}
	\itemsep0em 
		\item[(A1)]{AssoziativitûÊt: $x + (y + z) = (x + y) + z$}
		\item[(A2)]{Neutrales Element: $x + 0 = x$}
		\item[(A3)]{Inverses Element: $\forall x \in \R$ $\exists y \in \R :x + y = 0$}
		\item[(A4)]{KommutativitûÊt: $x + y = y + x$}
	\end{itemize}
	\item[(M)]{\textbf{Axiome der Multiplikation}}
	\begin{itemize}
	\itemsep0em 
		\item[(M1)]{AssoziativitûÊt: $x \cdot (y \cdot z) = (x \cdot y) \cdot z$}
		\item[(M2)]{Neutrales Element: $x \cdot 1 = x$}
		\item[(M3)]{Inverses Element: $\forall x \in \R$, $x \neq 0$ $\exists y \in \R :x \cdot y = 1$}
		\item[(M4)]{KommutativitûÊt: $x \cdot y = y \cdot x$}
	\end{itemize}
	\item[(D)]{\textbf{Axiom der DistributivitûÊt} $x \cdot (y + z) = x \cdot y + x \cdot z$}
	\item[(O)]{\textbf{Ordnungsaxiome} (1-3 Partialorder)}
	\begin{itemize}
	\itemsep0em 
		\item[(O1)]{ReflexivitûÊt: $x \leq x$ $\forall x \in \R$}
		\item[(O2)]{TransitivitûÊt: $x \leq y$ und $y \leq z \Rightarrow x \leq z$}
		\item[(O3)]{Antisymmetrie: $x \leq y$ und $y \leq x \Rightarrow x = y$}
		\item[(O4)]{Total: $\forall x,y \in \R$ gilt entweder $y \leq x$ oder $x \leq y$}
	\end{itemize}
	\item[(K)]{\textbf{KompatibilitûÊt}} 
	\begin{itemize}
	\itemsep0em 
		\item[(K1)]{$\forall x, y, z \in \R : x \leq y \Rightarrow x + z \leq y + z$ }
		\item[(K2)]{$\forall x \geq 0, \forall y \geq 0 : x \cdot y \geq 0$}
	\end{itemize}	
	\item[(V)]{\textbf{OrdnungsvollstûÊndigkeit}}
	Seien $A, B$ Teilmengen von $\R$, so dass
	\begin{itemize}
	\itemsep0em 
		\item[(i)] $A \neq \emptyset$, $B \neq \emptyset$
		\item[(ii)] $\forall a \in A$ und $\forall b \in B$ gilt: $a \leq b$  
	\end{itemize}
	Dann gibt es $c \in \R$, so dass $a \leq c \leq b$.
	\begin{center} 
		\includegraphics[width=\columnwidth/2]{ordnungsvoll.jpg}
	\end{center}
\end{itemize}
\end{multicols}

\Title{Supremum und Infimum}

\Def $A \subseteq \R$ heisst von oben [unten] beschrûÊnkt (v.o.b. [v.u.b.]), falls es $x \in \R$ gibt,
sodass $x \geq a$ [$x \leq a$] fû¥r alle $a \in A$. $x$ heiût dann obere [untere] Schranke von $A$. $A$ 
heisst beschrûÊnkt, falls $A$ v.o.b. und v.u.b. ist.

Falls $A$ eine obere Schranke mit $x \in A$ hat, ist $x$ das \textbf{Maximum} von $A$, 
$x = \Max(A)$. [Minimum analog]

\begin{tcolorbox}
	\Satz Falls $A \subseteq \R, A \neq \emptyset$ v.o.b. [v.u.b.] ist, gibt es eine kleinste obere 
	Schranke [grûÑsste untere Schranke] $x$ von $A$. $x$ heiût dann \textbf{Supremum [Infimum]}  
	von $A$, $x = \Sup(A)$ [$x = \Inf(A)$].
\end{tcolorbox}

Wenn \textbf{$x = \Sup(A)$} so gilt: 
\begin{enumerate}[label = (\arabic*)]
	\item{\(x\geq a\) fû¥r alle \(a\in A\)}
	\item{fû¥r alle \(y<x\) ist \(y\) keine obere Schranke von \(A\)\\
			   \(\Leftrightarrow\) fû¥r alle \(y<x\) gibt es \(a\in A\) mit \(y<a\)}
\end{enumerate}

Falls $A$ ein Maximum besitzt, ist $\Max(A) = \Sup(A)$. \medskip

\textbf{Eigenschaften}
\begin{enumerate}[label = (\arabic*)]
	\item{Wenn $A \subseteq B \subseteq \R$ und $B$ beschrûÊnkt ist, so ist $A$ beschrûÊnkt und $\Sup(A) 
	\leq \Sup(B), \Inf(A) \geq \Inf(B)$}
	\item{Wenn $A, B \subseteq \R, A, B \neq \emptyset$ und $a \leq b$ fû¥r alle $a \in A, b \in B$, dann 
	ist $A$ v.o.b, $B$ v.u.b. und $\Sup(A) \leq \Inf(B)$}
	\item[]{Notation: Fû¥r $A \neq \emptyset$ und nicht v.o.b. definieren wir $\Sup(A) = \infty$, 
	falls nicht v.u.b. $\Inf(A) = -\infty$}
\end{enumerate}
